# kOS Yachay Dockerfile and Deploy Script

# Dockerfile
FROM node:20-alpine

# Set working directory
WORKDIR /app

# Copy files
COPY . .

# Install dependencies
RUN npm install

# Expose REST API port
EXPOSE 3000

# Start the REST API server
CMD ["node", "./scripts/yachay-rest-server.js"]
